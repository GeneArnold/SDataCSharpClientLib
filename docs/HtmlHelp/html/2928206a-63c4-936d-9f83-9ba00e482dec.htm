Examples
CopyThe following code example demonstrates the usage of the SDataService class.
SDataService service = new SDataService();

// set user name to authenticate with
service.UserName = "lee";
// set password to authenticate with
service.Password = "";

service.Protocol = "HTTP";
service.ServerName = "sdata.acme.com";
service.ApplicationName = "sageApp";
service.VirtualDirectory = "sdata";
service.ContractName = "test";
service.DataSet = "-";

// or create from url
service = new SDataService("http://sdata.acme.com/sdata/sageApp/test/-/", "lee", "");

service.Initialize();
