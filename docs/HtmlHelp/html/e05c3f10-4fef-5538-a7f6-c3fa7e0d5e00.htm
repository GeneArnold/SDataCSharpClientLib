CopyThe following code example demonstrates the usage of the SDataSingleResourceRequest class.
AtomEntry ReadSingleReourceEntry(SDataService service)
{
    // Read a Single Resource Entry
    SDataSingleResourceRequest request = new SDataSingleResourceRequest(service);
    request.ResourceKind = "accounts";
    request.ResourceSelector = "('A001')";
    // the above configuration generates  http://sdata.acme.com/sdata/sageApp/test/-/accounts('A001') 

    // read the entry from the server
    AtomEntry entry = request.Read();
    return entry;
}
