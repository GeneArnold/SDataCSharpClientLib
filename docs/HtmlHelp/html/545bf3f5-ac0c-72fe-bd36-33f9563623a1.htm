// Read a Resource Collection Feed
void ReadResourceCollectionWithReader(SDataService service)
{
    SDataResourceCollectionRequest request = new SDataResourceCollectionRequest(service);
    request.ResourceKind = "accounts";

    // pageing
    request.StartIndex = 21;
    request.ItemsPerPage = 10;

    // query
    request.QueryValues.Add("where", "accountid EQ '123456789abc'");
    request.QueryValues.Add("orderby", "'account'");

    // the above configuration generates http://sdata.acme.com/sdata/sageApp/test/prod/accounts?startIndex=21&count=10 
    // Read the feed from the server
    AtomFeed feed = request.Read();

    AtomFeedReader reader = new AtomFeedReader(feed);

    while(reader.Read())
    {
        AtomEntry entry = reader.Entry;
        // do something with the entry
    }


}
