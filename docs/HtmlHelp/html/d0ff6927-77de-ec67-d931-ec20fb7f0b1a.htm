CopyThe following code example demonstrates the usage of the IntermediateServicesRequest class.
AtomFeed ReadEnumerationOfServices(SDataService service)
{
    IntermediateServicesRequest request = new IntermediateServicesRequest(service);

    request.ResourceKind = string.Empty;
    // the above configuration generates http://sdata.acme.com/sdata/sageApp/test/$service

    // read the feed from the server
    AtomFeed feed = request.Read();


    // reconfigure and set the resource kind
    request.ResourceKind = "accounts";
    // the above configuration generates http://sdata.acme.com/sdata/sageApp/test/accounts/$service
    // read the feed from the server
    feed = request.Read();

    return feed;
}
